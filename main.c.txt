#include <stdio.h>
#include <stdlib.h>
#include <string.h>


//学生结构体
typedef struct Student
{
    int id;             //学号
    char name[20];      //姓名
    int age;            //年龄

    struct Student *next;   //指向下一个学生

}Student;


//创建头节点
Student *head = NULL;


//显示菜单
void menu()
{
    printf("====================\n");
    printf(" 学生信息管理系统\n");
    printf("====================\n");
    printf("1.添加学生信息\n");
    printf("2.浏览学生信息\n");
    printf("3.查询学生信息\n");
    printf("4.删除学生信息\n");
    printf("5.修改学生信息\n");
    printf("0.退出系统\n");
    printf("====================\n");
}
//浏览所有学生信息
void showStudent()
{
    Student *p = head;


    //判断链表是否为空
    if(p == NULL)
    {
        printf("暂无学生信息！\n");
        return;
    }


    printf("======学生信息列表======\n");


    //遍历链表
    while(p != NULL)
    {

        printf("学号：%d\n",p->id);

        printf("姓名：%s\n",p->name);

        printf("年龄：%d\n",p->age);


        printf("----------------\n");


        //移动到下一个节点
        p = p->next;
    }

}
//添加学生信息
void addStudent()
{
    //创建新节点
    Student *newStudent;

    newStudent = (Student*)malloc(sizeof(Student));


    //输入学生信息
    printf("请输入学生学号：");
    scanf("%d",&newStudent->id);


    printf("请输入学生姓名：");
    scanf("%s",newStudent->name);


    printf("请输入学生年龄：");
    scanf("%d",&newStudent->age);



    //新节点指向空
    newStudent->next = NULL;



    //如果链表为空
    if(head == NULL)
    {
        head = newStudent;
    }
    else
    {
        //找到链表尾部
        Student *p = head;

        while(p->next != NULL)
        {
            p = p->next;
        }


        //尾部连接新节点
        p->next = newStudent;
    }


    printf("学生信息添加成功！\n");

}


int main()
{
    int choice;


    while(1)
    {

        menu();

        printf("请输入选择:");
        scanf("%d",&choice);


        switch(choice)
        {

            case 1:
                addStudent( );
                break;


            case 2:
                showStudent();
                break;


            case 3:
                printf("查询学生信息\n");
                break;


            case 4:
                printf("删除学生信息\n");
                break;


            case 5:
                printf("修改学生信息\n");
                break;


            case 0:
                printf("退出系统\n");
                return 0;


            default:
                printf("输入错误\n");

        }

    }


    return 0;
}
