#include <stdio.h>

int main()
{
    int choice;

    printf("===== 学生信息管理系统 =====\n");
    printf("1. 添加学生信息\n");
    printf("2. 浏览学生信息\n");
    printf("3. 查询学生信息\n");
    printf("0. 退出系统\n");

    printf("请输入选择：");
    scanf("%d",&choice);

    switch(choice)
    {
        case 1:
            printf("添加学生信息\n");
            break;

        case 2:
            printf("浏览学生信息\n");
            break;

        case 3:
            printf("查询学生信息\n");
            break;

        case 0:
            printf("退出系统\n");
            break;

        default:
            printf("输入错误\n");
    }

    return 0;
}
