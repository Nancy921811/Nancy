#include <stdio.h>
#include <stdlib.h>
#include <string.h>


//学生结构体
typedef struct Student
{
    int id;             //学号
    char name[20];      //姓名
    int age;            //年龄

    struct Student *next;   //指向下一个学生

}Student;


//创建头节点
Student *head = NULL;


//显示菜单
void menu()
{
    printf("====================\n");
    printf(" 学生信息管理系统\n");
    printf("====================\n");
    printf("1.添加学生信息\n");
    printf("2.浏览学生信息\n");
    printf("3.查询学生信息\n");
    printf("4.删除学生信息\n");
    printf("5.修改学生信息\n");
    printf("0.退出系统\n");
    printf("====================\n");
}


int main()
{
    int choice;


    while(1)
    {

        menu();

        printf("请输入选择:");
        scanf("%d",&choice);


        switch(choice)
        {

            case 1:
                printf("添加学生信息\n");
                break;


            case 2:
                printf("浏览学生信息\n");
                break;


            case 3:
                printf("查询学生信息\n");
                break;


            case 4:
                printf("删除学生信息\n");
                break;


            case 5:
                printf("修改学生信息\n");
                break;


            case 0:
                printf("退出系统\n");
                return 0;


            default:
                printf("输入错误\n");

        }

    }


    return 0;
}
